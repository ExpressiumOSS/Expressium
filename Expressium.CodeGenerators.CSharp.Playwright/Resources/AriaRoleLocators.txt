

// Standart web elements with ARIA roles and lables...
How=Role Using=Label
private WebTextBox FirstName => new WebTextBox(page, page.GetByRole(AriaRole.Textbox, new() { Name = "First Name" }));
private WebComboBox Country => new WebComboBox(page, page.GetByRole(AriaRole.Combobox, new() { Name = "Country" }));
private WebRadioButton Female => new WebRadioButton(page, page.GetByRole(AriaRole.Radio, new() { Name = "Female" }));
private WebCheckBox IAgreeToTheTermsOfUse => new WebCheckBox(page, page.GetByRole(AriaRole.Checkbox, new() { Name = "I agree to the Terms of Use..." }));
private WebButton Submit => new WebButton(page, page.GetByRole(AriaRole.Button, new() { Name = "Submit" }));
private WebLink CreateAnAccount => new WebLink(page, page.GetByRole(AriaRole.Link, new() { Name = "Create an account" }));

// Standart web elements without lables...
private WebTextBox FirstName => new WebTextBox(page, page.Locator("#yusername"));
private WebTextBox Password => new WebTextBox(page, page.Locator("[name='password']"));

<!-- IMPLICIT LABEL (not recommended for use) - the label wraps the input. -->
<label> 
  Name:
  <input type="text" name="name" />
</label>

<!-- EXPLICIT LABEL (recommended for use) - the label is connected to an input via "for" and "id" -->
<label for="explicit-label-name">Name: </label>
<input type="text" id="explicit-label-name" name="name" />
